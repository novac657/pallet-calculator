        <button onclick="calculatePalettes()">Calculate</button>

        <div class="result" id="result"></div>

    <script>
        function calculatePalettes() {
            const totalBoxes = parseInt(document.getElementById("totalBoxes").value);
            const minBoxes = parseInt(document.getElementById("minBoxes").value);
            const maxBoxes = parseInt(document.getElementById("maxBoxes").value);
            const resultDiv = document.getElementById("result");

            if (isNaN(totalBoxes) || isNaN(minBoxes) || isNaN(maxBoxes) || totalBoxes <= 0 || minBoxes <= 0 || maxBoxes < minBoxes) {
                resultDiv.innerHTML = "Invalid input. Please check your numbers.";
                return;
            }

            let maxPalettes = Math.floor(totalBoxes / maxBoxes);
            let remainder = totalBoxes % maxBoxes;

            if (remainder !== 0 && remainder < minBoxes) {
                const extraBoxesNeeded = minBoxes - remainder;
                maxPalettes -= Math.ceil(extraBoxesNeeded / maxBoxes);
                remainder = totalBoxes - maxPalettes * maxBoxes;
            }

            const minPalettes = remainder > 0 ? Math.ceil(remainder / minBoxes) : 0;
            const totalPalettes = maxPalettes + minPalettes;

            resultDiv.innerHTML = `
                <p>Pallets with max boxes: ${maxPalettes}</p>
                <p>Pallets with min boxes: ${minPalettes}</p>
                <p>Total pallets: ${totalPalettes}</p>
            `;
        }
    </script>
